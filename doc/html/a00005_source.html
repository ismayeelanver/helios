166void WindowManager::tile_windows() {
167 int num_windows = windows.size();
168 if (num_windows == 0)
169 return;
170
171 int screen_width = screen->width_in_pixels;
172 int screen_height = screen->height_in_pixels;
173 int gap = config.window.gap;
174 int border_width = config.border.width;
175
176 if (num_windows == 1) {
177 uint32_t values[4] = {static_cast<uint32_t>(gap),
178 static_cast<uint32_t>(gap),
179 static_cast<uint32_t>(screen_width - 2 * gap),
180 static_cast<uint32_t>(screen_height - 2 * gap)};
181
182 uint32_t border_values[1] = {static_cast<uint32_t>(border_width)};
183 xcb_configure_window(conn, windows[0], XCB_CONFIG_WINDOW_BORDER_WIDTH,
184 border_values);
185 xcb_configure_window(conn, windows[0],
186 XCB_CONFIG_WINDOW_X | XCB_CONFIG_WINDOW_Y |
187 XCB_CONFIG_WINDOW_WIDTH | XCB_CONFIG_WINDOW_HEIGHT,
188 values);
189 xcb_set_input_focus(conn, XCB_INPUT_FOCUS_POINTER_ROOT, windows[0],
190 XCB_CURRENT_TIME);
191 xcb_flush(conn);
192 return;
193 }
194
195 struct WindowRect {
196 int x, y, width, height;
197 };
198 std::vector<WindowRect> window_rects(num_windows);
199
200 window_rects[0] = {gap, gap, screen_width - 2 * gap, screen_height - 2 * gap};
201
202 bool split_horizontal = false;
203 for (int i = 1; i < num_windows; i++) {
204
205 int largest_idx = i - 1;
206 int largest_area = 0;
207 for (int j = 0; j < i; j++) {
208 int area = window_rects[j].width * window_rects[j].height;
209 if (area > largest_area) {
210 largest_area = area;
211 largest_idx = j;
212 }
213 }
214
215 WindowRect &to_split = window_rects[largest_idx];
216 WindowRect &new_window = window_rects[i];
217
218 if (split_horizontal) {
219
220 int new_height = (to_split.height - gap) / 2;
221 new_window = {to_split.x, to_split.y + new_height + gap, to_split.width,
222 to_split.height - new_height - gap};
223 to_split.height = new_height;
224 } else {
225
226 int new_width = (to_split.width - gap) / 2;
227 new_window = {to_split.x + new_width + gap, to_split.y,
228 to_split.width - new_width - gap, to_split.height};
229 to_split.width = new_width;
230 }
231
232 split_horizontal = !split_horizontal;
233 }
234
235 for (int i = 0; i < num_windows; i++) {
236 const auto &rect = window_rects[i];
237 uint32_t values[4] = {
238 static_cast<uint32_t>(rect.x), static_cast<uint32_t>(rect.y),
239 static_cast<uint32_t>(rect.width), static_cast<uint32_t>(rect.height)};
240
241 uint32_t border_values[1] = {static_cast<uint32_t>(border_width)};
242 xcb_configure_window(conn, windows[i], XCB_CONFIG_WINDOW_BORDER_WIDTH,
243 border_values);
244 xcb_configure_window(conn, windows[i],
245 XCB_CONFIG_WINDOW_X | XCB_CONFIG_WINDOW_Y |
246 XCB_CONFIG_WINDOW_WIDTH | XCB_CONFIG_WINDOW_HEIGHT,
247 values);
248 xcb_set_input_focus(conn, XCB_INPUT_FOCUS_POINTER_ROOT, windows[i],
249 XCB_CURRENT_TIME);
250 }
251
252 xcb_flush(conn);
253}

348void WindowManager::handle_map_request(xcb_generic_event_t *ev) {
349 auto event = (xcb_map_request_event_t *)ev;
350 auto window = event->window;
351 windows.push_back(window);
352
353 auto cookie = xcb_get_window_attributes(conn, window);
354 xcb_generic_error_t **e = nullptr;
355 auto _attr_reply = xcb_get_window_attributes_reply(conn, cookie, e);
356
357 if (_attr_reply->override_redirect) {
358 delete _attr_reply;
359 delete e;
360 return;
361 }
362
363 uint32_t values[] = {XCB_EVENT_MASK_ENTER_WINDOW |
364 XCB_EVENT_MASK_FOCUS_CHANGE |
365 XCB_EVENT_MASK_PROPERTY_CHANGE};
366 xcb_change_window_attributes(conn, window, XCB_CW_EVENT_MASK, values);
367
368 xcb_map_window(conn, window);
369
370 auto border_width = static_cast<uint32_t>(config.border.width);
371 uint32_t border_values[2] = {border_width, border_width};
372
373 xcb_configure_window(conn, window, XCB_CONFIG_WINDOW_BORDER_WIDTH,
374 border_values);
375
376 set_window_border_color(window, config.border.inactive_color);
377
378 update_focus(window);
379
380 xcb_flush(conn);
381 tile_windows();
382
383 delete _attr_reply;
384 delete e;
385}

482void WindowManager::run() {
483 for (;;) {
484 xcb_generic_event_t *event = xcb_wait_for_event(conn);
485
486 if (!event) {
487 logger->error("Event is invalid");
488 break;
489 }
490
491 auto ev = event->response_type & ~0x80;
492 auto handler = evH.find(ev);
493 if (handler != evH.end()) {
494 handler->second((xcb_generic_event_t *)event);
495 } else {
496 logger->error("Invalid Event Type!");
497 }
498 delete event;
499 }
