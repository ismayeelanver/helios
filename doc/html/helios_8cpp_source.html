159void WindowManager::tile_windows() {
160 int num_windows = windows.size();
161 if (num_windows == 0)
162 return;
163
164 int screen_width = screen->width_in_pixels;
165 int screen_height = screen->height_in_pixels;
166 int gap = config.window.gap;
167 int border_width = config.border.width;
168
169 if (num_windows == 1) {
170 uint32_t values[4] = {static_cast<uint32_t>(gap),
171 static_cast<uint32_t>(gap),
172 static_cast<uint32_t>(screen_width - 2 * gap),
173 static_cast<uint32_t>(screen_height - 2 * gap)};
174
175 uint32_t border_values[1] = {static_cast<uint32_t>(border_width)};
176 xcb_configure_window(conn, windows[0], XCB_CONFIG_WINDOW_BORDER_WIDTH,
177 border_values);
178 xcb_configure_window(conn, windows[0],
179 XCB_CONFIG_WINDOW_X | XCB_CONFIG_WINDOW_Y |
180 XCB_CONFIG_WINDOW_WIDTH | XCB_CONFIG_WINDOW_HEIGHT,
181 values);
182 set_window_border_color(windows[0], config.border.color);
183 xcb_flush(conn);
184 return;
185 }
186
187 struct WindowRect {
188 int x, y, width, height;
189 };
190 std::vector<WindowRect> window_rects(num_windows);
191
192 window_rects[0] = {gap, gap, screen_width - 2 * gap, screen_height - 2 * gap};
193
194 bool split_horizontal = false;
195 for (int i = 1; i < num_windows; i++) {
196
197 int largest_idx = i - 1;
198 int largest_area = 0;
199 for (int j = 0; j < i; j++) {
200 int area = window_rects[j].width * window_rects[j].height;
201 if (area > largest_area) {
202 largest_area = area;
203 largest_idx = j;
204 }
205 }
206
207 WindowRect &to_split = window_rects[largest_idx];
208 WindowRect &new_window = window_rects[i];
209
210 if (split_horizontal) {
211
212 int new_height = (to_split.height - gap) / 2;
213 new_window = {to_split.x, to_split.y + new_height + gap, to_split.width,
214 to_split.height - new_height - gap};
215 to_split.height = new_height;
216 } else {
217
218 int new_width = (to_split.width - gap) / 2;
219 new_window = {to_split.x + new_width + gap, to_split.y,
220 to_split.width - new_width - gap, to_split.height};
221 to_split.width = new_width;
222 }
223
224 split_horizontal = !split_horizontal;
225 }
226
227 for (int i = 0; i < num_windows; i++) {
228 const auto &rect = window_rects[i];
229 uint32_t values[4] = {
230 static_cast<uint32_t>(rect.x), static_cast<uint32_t>(rect.y),
231 static_cast<uint32_t>(rect.width), static_cast<uint32_t>(rect.height)};
232
233 uint32_t border_values[1] = {static_cast<uint32_t>(border_width)};
234 xcb_configure_window(conn, windows[i], XCB_CONFIG_WINDOW_BORDER_WIDTH,
235 border_values);
236 xcb_configure_window(conn, windows[i],
237 XCB_CONFIG_WINDOW_X | XCB_CONFIG_WINDOW_Y |
238 XCB_CONFIG_WINDOW_WIDTH | XCB_CONFIG_WINDOW_HEIGHT,
239 values);
240 set_window_border_color(windows[i], config.border.color);
241 }
242
243 xcb_flush(conn);
244}

320void WindowManager::handle_map_request(xcb_window_t window) {
321 windows.push_back(window);
322
323 auto cookie = xcb_get_window_attributes(conn, window);
324 xcb_generic_error_t **e = nullptr;
325 auto _attr_reply = xcb_get_window_attributes_reply(conn, cookie, e);
326
327 if (_attr_reply->override_redirect) {
328 return;
329 }
330
331 delete e;
332
333 xcb_map_window(conn, window);
334
335 auto border_width = static_cast<uint32_t>(config.border.width);
336 uint32_t border_values[2] = {border_width, border_width};
337
338 xcb_configure_window(conn, window, XCB_CONFIG_WINDOW_BORDER_WIDTH,
339 border_values);
340 xcb_flush(conn);
341
342 set_window_border_color(window, config.border.color);
343
344 xcb_set_input_focus(conn, XCB_INPUT_FOCUS_POINTER_ROOT, window,
345 XCB_CURRENT_TIME);
346 xcb_flush(conn);
347 tile_windows();

424void WindowManager::run() {
425 for (;;) {
426 xcb_generic_event_t *event = xcb_wait_for_event(conn);
427 logger->info("Event Happened!");
428
429 if (!event) {
430 logger->error("Event is invalid");
431 break;
432 }
433
434 switch (event->response_type & ~0x80) {
435 case XCB_MAP_REQUEST: {
436 logger->info("found map-request event");
437 auto *map_request = (xcb_map_request_event_t *)event;
438 handle_map_request(map_request->window);
439 break;
440 }
441 case XCB_UNMAP_NOTIFY: {
442 logger->info("found unmap-notify event");
443 auto *unmap_notify = (xcb_unmap_notify_event_t *)event;
444 handle_unmap_request(unmap_notify->window);
445 break;
446 }
447 case XCB_DESTROY_NOTIFY: {
448 logger->info("found destory-notify event");
449 auto destroy_notify = (xcb_destroy_notify_event_t *)event;
450 handle_destroy_notify(destroy_notify->window);
451 break;
452 }
453 case XCB_ENTER_NOTIFY: {
454 logger->info("found enter-notify event");
455 auto enter_notify = (xcb_enter_notify_event_t *)event;
456 set_focus(enter_notify->root);
457 break;
458 }
459 case XCB_KEY_PRESS: {
460 logger->info("found key-press event");
461 auto key_press = (xcb_key_press_event_t *)event;
462 handle_key_press(key_press);
463 break;
464 }
465 }
466
467 free(event);
468 xcb_flush(conn);
469 }
